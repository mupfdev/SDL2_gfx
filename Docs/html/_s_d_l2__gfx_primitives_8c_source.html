01036                 * that we won't overrun the endpoint along the X axis. 
01037                 */

01039                 * Not-so-portable version: erradj = ((Uint64)dy << 32) / (Uint64)dx; 
01040                 */
01041                 erradj = ((dy << 16) / dx) << 16;

01044                 * draw all pixels other than the first and last

01053                                 * Accumulator turned over, advance y 
01054                                 */

01056                                 y0p1++;
01057                         }
01058                         xx0 += xdir;    /* x-major so always advance X */
01059                         /*
01060                         * the AAbits most significant bits of erracc give us the intensity
01061                         * weighting for this pixel, and the complement of the weighting for
01062                         * the paired pixel. 
01063                         */

01065                         result |= pixelRGBAWeight (renderer, xx0, yy0, r, g, b, a, 255 - wgt);
01066                         result |= pixelRGBAWeight (renderer, xx0, y0p1, r, g, b, a, wgt);

01072         */

01076                 * need to be weighted. 
01077                 */
01078                 result |= pixelRGBA (renderer, x2, y2, r, g, b, a);
01079         }
01080 
01081         return (result);

01339                 }

01358                         xpcx = x + cx;
01359                         xmcx = x - cx;

01361                         // always check if we're drawing a certain octant before adding a pixel to that octant.
01362                         if (drawoct & 4)  result |= pixel(renderer, xmcx, ypcy);

01365                         if (drawoct & 64) result |= pixel(renderer, xpcx, ymcy);
01366                 } else {
01367                         if (drawoct & 96) result |= pixel(renderer, x, ymcy);
01368                         if (drawoct & 6)  result |= pixel(renderer, x, ypcy);
01369                 }
01370

01372                 xmcy = x - cy;
01373                 if (cx > 0 && cx != cy) {
01374                         ypcx = y + cx;
01375                         ymcx = y - cx;
01376                         if (drawoct & 8)   result |= pixel(renderer, xmcy, ypcx);
01377                         if (drawoct & 1)   result |= pixel(renderer, xpcy, ypcx);
01378                         if (drawoct & 16)  result |= pixel(renderer, xmcy, ymcx);

01392                         else                                               drawoct |= (1 << startoct);
01393                 }

01401                 */

01403                         df += d_e;
01404                         d_e += 2;
01405                         d_se += 2;
01406                 } else {
01407                         df += d_se;
01408                         d_e += 2;
01409                         d_se += 4;
01410                         cy--;
01411                 }

01413         } while (cx <= cy);
01414 
01415         return (result);

01528 int _ellipseRGBA(SDL_Renderer * renderer, Sint16 x, Sint16 y, Sint16 rx, Sint16 ry, Uint8 r, Uint8 g, Uint8 b, Uint8 a, Sint32 f)
01529 {
01530         int result;
01531         Sint32 rx2, ry2, rx22, ry22; 
01532     Sint32 error;
01533     Sint32 curX, curY, curXp1, curYm1;
01534         Sint32 scrX, scrY, oldX, oldY;
01535     Sint32 deltaX, deltaY;
01536 
01537         /*
01538         * Sanity check radii 
01539         */
01540         if ((rx < 0) || (ry < 0)) {
01541                 return (-1);
01542         }
01543 
01544         /*
01545         * Set color
01546         */
01547         result = 0;
01548         result |= SDL_SetRenderDrawBlendMode(renderer, (a == 255) ? SDL_BLENDMODE_NONE : SDL_BLENDMODE_BLEND);
01549         result |= SDL_SetRenderDrawColor(renderer, r, g, b, a);
01550 
01551         /*
01552         * Special cases for rx=0 and/or ry=0: draw a hline/vline/pixel 
01553         */
01554         if (rx == 0) {
01555                 if (ry == 0) {
01556                         return (pixel(renderer, x, y));
01557                 } else {
01558                         return (vline(renderer, x, y - ry, y + ry));
01559                 }
01560         } else {
01561                 if (ry == 0) {
01562                         return (hline(renderer, x - rx, x + rx, y));
01563                 }
01564         }
01565 
01566         /*
01567          * Top/bottom center points.
01568          */
01569         oldX = scrX = 0;
01570         oldY = scrY = ry;
01571         result |= _drawQuadrants(renderer, x, y, 0, ry, f);
01572 
01573         /* Midpoint ellipse algorithm with overdraw */
01574         rx *= ELLIPSE_OVERSCAN;
01575         ry *= ELLIPSE_OVERSCAN;
01576         rx2 = rx * rx;
01577         rx22 = rx2 + rx2;
01578     ry2 = ry * ry;
01579         ry22 = ry2 + ry2;
01580     curX = 0;
01581     curY = ry;
01582     deltaX = 0;
01583     deltaY = rx22 * curY;
01584  
01585         /* Points in segment 1 */ 
01586     error = ry2 - rx2 * ry + rx2 / 4;
01587     while (deltaX <= deltaY)
01588     {
01589           curX++;
01590           deltaX += ry22;
01591  
01592           error +=  deltaX + ry2; 
01593           if (error >= 0)
01594           {
01595                curY--;
01596                deltaY -= rx22; 
01597                error -= deltaY;
01598           }
01599 
01600                   scrX = curX/ELLIPSE_OVERSCAN;
01601                   scrY = curY/ELLIPSE_OVERSCAN;
01602                   if ((scrX != oldX && scrY == oldY) || (scrX != oldX && scrY != oldY)) {
01603                         result |= _drawQuadrants(renderer, x, y, scrX, scrY, f);
01604                         oldX = scrX;
01605                         oldY = scrY;
01606                   }
01607     }
01608 
01609         /* Points in segment 2 */
01610         if (curY > 0) 
01611         {
01612                 curXp1 = curX + 1;
01613                 curYm1 = curY - 1;
01614                 error = ry2 * curX * curXp1 + ((ry2 + 3) / 4) + rx2 * curYm1 * curYm1 - rx2 * ry2;
01615                 while (curY > 0)
01616                 {
01617                         curY--;
01618                         deltaY -= rx22;
01619 
01620                         error += rx2;
01621                         error -= deltaY;
01622  
01623                         if (error <= 0) 
01624                         {
01625                curX++;
01626                deltaX += ry22;
01627                error += deltaX;
01628                         }
01629 
01630                     scrX = curX/ELLIPSE_OVERSCAN;
01631                     scrY = curY/ELLIPSE_OVERSCAN;
01632                     if ((scrX != oldX && scrY == oldY) || (scrX != oldX && scrY != oldY)) {
01633                                 oldY--;
01634                                 for (;oldY >= scrY; oldY--) {
01635                                         result |= _drawQuadrants(renderer, x, y, scrX, oldY, f);
01636                                         /* prevent overdraw */
01637                                         if (f) {
01638                                                 oldY = scrY - 1;
01639                                         }
01640                                 }
01641                                 oldX = scrX;
01642                                 oldY = scrY;
01643                     }           
01644                 }
01645 
01646                 /* Remaining points in vertical */
01647                 if (!f) {
01648                         oldY--;
01649                         for (;oldY >= 0; oldY--) {
01650                                 result |= _drawQuadrants(renderer, x, y, scrX, oldY, f);
01651                         }
01652                 }
01653         }
01654 
01655         return (result);
01656 }

01811 {

01815         Sint16 xp, yp, xs, ys, dyt, od, xx, yy, xc2, yc2;
01816         float cp;
01817         double sab;
01818         Uint8 weight, iweight;
01819 
01820         /*
01821         * Sanity check radii 
01822         */
01823         if ((rx < 0) || (ry < 0)) {
01824                 return (-1);
01825         }
01826 
01827         /*
01828         * Special cases for rx=0 and/or ry=0: draw a hline/vline/pixel 
01829         */
01830         if (rx == 0) {
01831                 if (ry == 0) {
01832                         return (pixelRGBA(renderer, x, y, r, g, b, a));
01833                 } else {
01834                         return (vlineRGBA(renderer, x, y - ry, y + ry, r, g, b, a));
01835                 }

01837                 if (ry == 0) {
01838                         return (hlineRGBA(renderer, x - rx, x + rx, y, r, g, b, a));
01839                 }
01840         }
01841 
01842         /* Variable setup */
01843         a2 = rx * rx;
01844         b2 = ry * ry;
01845 
01846         ds = 2 * a2;
01847         dt = 2 * b2;
01848 
01849         xc2 = 2 * x;
01850         yc2 = 2 * y;
01851 
01852         sab = sqrt((double)(a2 + b2));
01853         od = (Sint16)lrint(sab*0.01) + 1; /* introduce some overdraw */
01854         dxt = (Sint16)lrint((double)a2 / sab) + od;
01855 
01856         t = 0;
01857         s = -2 * a2 * ry;
01858         d = 0;

01861         yp = y - ry;

01877                 if (d >= 0)

01880                         if ((2 * d - s - a2) >= 0)
01881                                 ys = yp + 1;
01882                         else {
01883                                 ys = yp;
01884                                 yp++;
01885                                 d -= s + a2;
01886                                 s += ds;
01887                         }

01890                         ys = yp + 1;

01892                         s += ds;

01895                 t -= dt;

01943                                 xp--;
01944                                 d += t - b2;

01947                 } else {
01948                         xp--;

01950                         d += t - b2;

01961                         }
01962                 } else {
01963                         cp = 1.0;

01968                 iweight = 255 - weight;
01969 
01970                 /* Left half */
01971                 xx = xc2 - xp;

03559                 return(data[ndata-1]);

03566         n=ndata-1;

03570         for (k=0;k<=n;k++) {
03571                 nn = n;
03572                 kn = k;
03573                 nkn = n - k;
03574                 blend = muk * munk;

03576                 munk /= (1-mu);
03577                 while (nn >= 1) {

03581                                 blend /= (double)kn;
03582                                 kn--;
03583                         }

03585                                 blend /= (double)nkn;

03588                 }
